    def get_context_for_step(self, step, reference_number=None):
        context = {'step': step, 'reference_number': reference_number}
        if reference_number:
            temp_form = TemporaryFormData.objects.filter(reference_number=reference_number).first()
            if temp_form:
                if step == 1:
                    context['form'] = ContactForm(initial={
                        'email': temp_form.email,
                        'phone_number': temp_form.phone_number,
                        'first_name': temp_form.first_name,
                        'last_name': temp_form.last_name,
                        'report_type': temp_form.report_type,
                        'existing_reporting_company': temp_form.existing_reporting_company,
                    })
                elif step == 2:
                    context['form'] = ReportingCompanyForm(initial={
                        'sunbiz_report':temp_form.sunbiz_report,
                        'request_fincen_id':temp_form.request_fincen_id,
                        'foreign_pooled_investment_vehicle':temp_form.foreign_pooled_investment_vehicle,
                        'reporting_company_legal_name':temp_form.reporting_company_legal_name,
                        'alternate_name':temp_form.alternate_name,
                        'tax_identification_type':temp_form.tax_identification_type,
                        'tax_id_number':temp_form.tax_id_number,
                        'country_jurisdiction':temp_form.country_jurisdiction,
                        'state_of_information':temp_form.state_of_information,
                        'tribal_jurisdiction_of_formation':temp_form.tribal_jurisdiction_of_formation,
                        'address':temp_form.address,
                        'city':temp_form.city,
                        'us_or_us_territory':temp_form.us_or_us_territory,
                        'state':temp_form.state,
                        'zip_code':temp_form.zip_code, 
                    })
                elif step == 3:
                    bo_data = json.loads(temp_form.bo_data or '[]')
                    context['formset'] = BeneficialOwnerFormSet(initial=bo_data, prefix='bo')
                elif step == 4:
                    ca_data = json.loads(temp_form.ca_data or '[]')
                    context['formset'] = CompanyApplicantFormSet(initial=ca_data, prefix='ca')
        else:
            if step == 1:
                context['form'] = ContactForm()
            elif step == 2:
                context['form'] = ReportingCompanyForm()
            elif step == 3:
                context['formset'] = BeneficialOwnerFormSet(prefix='bo')
            elif step == 4:
                context['formset'] = CompanyApplicantFormSet(prefix='ca')

        return context
